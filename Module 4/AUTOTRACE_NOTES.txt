SQL> SET AUTOTRACE OFF -default

SQL> SET AUTOTRACE ON - both execution plan and statistics

SQL> SET AUTOTRACE ON EXPLAIN - only execution plan


SQL> SET AUTOTRACE ON STATISTICS - only statistics


SQL> SET AUTOTRACE TRACEONLY 

SQL> SET AUTOTRACE TRACEONLY EXPLAIN

SQL> SET AUTOTRACE TRACEONLY STATISTICS

*WITH TRACEONLY THE ROWS(DATA) WON'T BE SHOWN
